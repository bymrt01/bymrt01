      width: 100%;

<body>

  <div id="chart_div"></div>

  <script type="text/javascript">
    google.charts.load('current', {
      'packages': ['timeline']
    });
    
    function drawChart() {
      console.log("drawChart called"); // log a message

      var opts = {
        sendMethod: 'auto'
      };
      // Replace the URL below with the URL of your Google Spreadsheet, including the gid parameter
      var query = new google.visualization.Query('https://docs.google.com/spreadsheets/d/11Lz2jIpLFlD_JxBzaKHmkqOZmin9SLSbKcen3Ole0HI/edit#gid=0', opts);
      query.send(handleQueryResponse);
    }

    function handleQueryResponse(response) {
      console.log(response); // log the response object

      if (response.isError()) {
        console.log('Error in query: ' + response.getMessage() + ' ' + response.getDetailedMessage());
        return;
      }

      var data = response.getDataTable();

      var chart = new google.visualization.Timeline(document.getElementById('chart_div'));
      chart.draw(data, {
        chartArea: {
          width: '100%',
          height: '100%'
        },
        colors: ['#C0C0C0', '#adc6e5', '#72aae4', '#518cc2', '#4A7FB0', '#113477'],
        timeline: {
          rowLabelStyle: {
            fontName: 'Tahoma',
            fontSize: 10,
            color: '#0000'
          },
          barLabelStyle: {
            fontName: 'Tahoma',
            fontSize: 10,
            color: '#FFFFFF'
          }
        },
        width: 1350,
        height: 900,
        avoidOverlappingGridLines: false,
        alternatingRowStyle: false,
        backgroundColor: '#FFFFFF'
      });
    }

    google.charts.setOnLoadCallback(drawChart);
//This script is written by Murat Can ERK. If you have any questions please contact me from muratcan.erk@gmail.com
  </script>
</body>
